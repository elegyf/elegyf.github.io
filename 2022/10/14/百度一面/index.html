for(var i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, 1000);
}  //5 5 5 5 5 隔1s输出

for(var i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, i*1000);
}  //5 5 5 5 5 每隔1s输出

for(var i = 0; i < 5; i++) {
  (function(i) {
    setTimeout(function() {
      console.log(i);
    }, i*1000);    
  })(i);
}   //0 1 2 3 4每隔1s输出一个

for(var i = 0; i < 5; i++) {
  (function() {
    setTimeout(function() {
      console.log(i);
    }, i*1000);    
  })(i);
} //5 5 5 5 5 每隔1s输出  闭包

for(var i = 0; i < 5; i++) {
    setTimeout((function(i) {
      console.log(i);
    })(i), i*1000);    
}//0 1 2 3 4一下子输出
